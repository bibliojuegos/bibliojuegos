//Juegos
create table juegos(
idJuego int identity(1,1),
nombre varchar(30),
idPlataforma int,
idioma varchar(20),
ano numeric(4),
restriccion varchar(20),
idCompania int,
idGenero int,
stock numeric(4),
precio numeric(6),
primary key(idJuego)
);

create table cliente
  (
    idCliente int identity(1,1) not null ,
    nombre    varchar (20) ,
    apellido  varchar (20) ,
    telefono varchar (20),
    usuario varchar(20),
    clave varchar(20),
    esAdmin char(1),
    primary key(idCliente)
  ) ;


create table compania
  (
    idCompania int identity (1,1) not null ,
    nombre     varchar (20) ,
    idPais     int,
    primary key(idCompania)
  ) ;


create table empleados
  (
    idEmpleado int identity(1,1) not null ,
    rut        varchar (15) ,
    nombre     varchar (20) ,
    apellido   varchar (20) ,
    telefono   varchar (15) ,
    idSucursal int,
    primary key(idEmpleado)
  ) ;


create table genero
  (
    idGenero int identity(1,1) not null ,
    nombre   varchar (20),
    primary key(idGenero)
  ) ;


create table pais
  (
    idPais int identity(1,1) not null ,
    nombre varchar (20),
    primary key(idPais)
  ) ;


create table plataforma
  (
    idPlataforma int identity(0,1) not null ,
    nombre       varchar (20) ,
    lanzamiento  numeric (4),
    primary key(idPlataforma)
  ) ;


create table ventas
  (
    idVenta    int identity(1,1) not null ,
    idCliente   int,
    idEmpleado  int,
    idSucursal  int,
    idJuego     int,
    totalCompra numeric (8),
    fechaCompra datetime,
    primary key(idVenta)
  ) ;


create table sucursal
  (
    idSucursal int identity(1,1) not null ,
    direccion  varchar (30) ,
    telefono   varchar (15) ,
    email      varchar (25),
    primary key(idSucursal)
  ) ;
