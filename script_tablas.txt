create table juegos(
idJuego numeric(3),
nombre varchar(30),
idPlataforma numeric(3),
idioma varchar(20),
ano numeric(4),
restriccion varchar(20),
idCompania numeric(3),
idGenero numeric(3),
stock numeric(4),
precio numeric(6),
primary key(idJuego)
);

CREATE
  TABLE Cliente
  (
    IdCliente NUMERIC (3) NOT NULL ,
    Nombre    VARCHAR (20) ,
    Apellido  VARCHAR (20) ,
    Edad      NUMERIC (2) ,
    IdVentas  NUMERIC (10)
  ) ;


CREATE TABLE Compañia
  (
    IdCompañia NUMERIC (3) NOT NULL ,
    Nombre     VARCHAR (20) ,
    Pais       NUMERIC (3) ,
    IdPais     NUMERIC (3)
  ) ;


CREATE TABLE Empleados
  (
    IdEmpleado NUMERIC (3) NOT NULL ,
    rut        VARCHAR (15) ,
    Nombre     VARCHAR (20) ,
    apellido   VARCHAR (20) ,
    telefono   VARCHAR (15) ,
    IdSucursal NUMERIC (3)
  ) ;


CREATE TABLE Genero
  (
    IdGenero NUMERIC (3) NOT NULL ,
    Nombre   VARCHAR (20)
  ) ;


CREATE  TABLE Pais
  (
    IdPais NUMERIC (3) NOT NULL ,
    Nombre VARCHAR (20)
  ) ;


CREATE TABLE Plataforma
  (
    IdPlataforma NUMERIC (3) NOT NULL ,
    Nombre       VARCHAR (20) ,
    Lanzamiento  NUMBERIC (4)
  ) ;


CREATE  TABLE Ventas
  (
    IdVentas    NUMERIC (10) NOT NULL ,
    Cliente     NUMERIC (3) ,
    sucursal    NUMERIC (3) ,
    juego       NUMERIC (3) ,
    IdSucursal  NUMERIC (3) ,
    TotalCompra NUMERIC (6)
  ) ;


CREATE TABLE sucursal
  (
    IdSucursal NUMERIC (3) NOT NULL ,
    direccion  VARCHAR (30) ,
    Telefono   VARCHAR (15) ,
    Email      VARCHAR (25)
  ) ;
