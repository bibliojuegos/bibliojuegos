//Juegos
create table juegos(
idJuego numeric(3),
nombre varchar(30),
idPlataforma numeric(3),
idioma varchar(20),
ano numeric(4),
restriccion varchar(20),
idCompania numeric(3),
idGenero numeric(3),
stock numeric(4),
precio numeric(6),
primary key(idJuego)
);

create table cliente
  (
    idCliente numeric (3) not null ,
    nombre    varchar (20) ,
    apellido  varchar (20) ,
    edad      numeric (2) ,
    idVentas  numeric (10),
    telefono varchar (20),
    primary key(idCliente)
  ) ;


create table compania
  (
    idCompania numeric (3) not null ,
    nombre     varchar (20) ,
    idPais     numeric (3),
    primary key(idCompania)
  ) ;


create table empleados
  (
    idEmpleado numeric (3) not null ,
    rut        varchar (15) ,
    nombre     varchar (20) ,
    apellido   varchar (20) ,
    telefono   varchar (15) ,
    idSucursal numeric (3),
    primary key(idEmpleado)
  ) ;


create table genero
  (
    idGenero numeric (3) not null ,
    nombre   varchar (20),
    primary key(idGenero)
  ) ;


create table pais
  (
    idPais numeric (3) not null ,
    nombre varchar (20),
    primary key(idPais)
  ) ;


create table plataforma
  (
    idPlataforma numeric (3) not null ,
    nombre       varchar (20) ,
    lanzamiento  numeric (4),
    primary key(idPlataforma)
  ) ;


create table ventas
  (
    idVenta    numeric (10) not null ,
    idCliente   numeric (3) ,
    idEmpleado  numeric(3),
    idSucursal  numeric (3) ,
    idJuego     numeric (3) ,
    totalCompra numeric (8),
    fechaCompra datetime,
    primary key(idVenta)
  ) ;


create table sucursal
  (
    idSucursal numeric (3) not null ,
    direccion  varchar (30) ,
    telefono   varchar (15) ,
    email      varchar (25),
    primary key(idSucursal)
  ) ;
